5.4 Creación de funciones con control de flujo

Tomaremos el siguiente ejemplo para crear una función que nos permita reemplazar los valores -999 por NA.

# Generar una base de datos aleatoria utilizando funciones básicas de R
set.seed(123)
DF <- data.frame(replicate(5, sample(c(seq(0.5, 7.9, length = 10), -999, 5555), 12, rep = TRUE)))
DF <- round(DF,2)
colnames(DF) <- paste0("Var", 1:5)

cambiarNA3 <- function(x, vector) {
  for (i in seq_along(vector)) {
  x[x == vector[i]] <- NA
  }
  resultados <<- x   
  print(x)
}

x_option <- function(x) {
  switch(x,
    a = "option 1",
    b = "option 2",
    c = "option 3",
    stop("Invalid `x` value")
  )
}
x_option("b")
